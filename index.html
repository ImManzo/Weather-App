// Weather App with city autocomplete + signed temps
const cities = ["Cairo","London","New York","Tokyo","Dubai","Istanbul",/*...*/];

function formatTemp(t) {
  const r = Math.round(t);
  return (r > 0 ? "+" : "") + r + "°C";
}

function showSuggestions(value) {
  const filtered = cities.filter(c =>
    c.toLowerCase().startsWith(value.toLowerCase()));
  sugg.innerHTML = filtered.map(c =>
    `<div onclick="selectCity('${c}')">${c}</div>`).join("");
  sugg.style.display = filtered.length ? "block" : "none";
}

async function getWeather() {
  const city = input.value.trim() || "London";
  const res = await fetch(
    `https://wttr.in/${encodeURIComponent(city)}?format=j1`);
  const data = await res.json();
  if (!data.current_condition) { showError("City not found"); return; }
  document.getElementById("cityName").textContent = data.nearest_area[0].areaName[0].value;
  document.getElementById("temp").textContent = formatTemp(data.current_condition[0].temp_C);
  document.getElementById("desc").textContent = data.current_condition[0].weatherDesc[0].value;
  document.getElementById("humidity").textContent = `${data.current_condition[0].humidity}%`;
  document.getElementById("wind").textContent = `${data.current_condition[0].windspeedKmph} km/h`;
  document.getElementById("feels").textContent = formatTemp(data.current_condition[0].FeelsLikeC);
}
